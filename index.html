My Simple Strava Dashboard

from pyscript import Element # Simulated Strava activities activities = [ {"name": "Morning Run", "distance": 5}, {"name": "Evening Ride", "distance": 20}, {"name": "Hike", "distance": 8} ] # Function to calculate total distance def total_distance(activities): return sum(a["distance"] for a in activities) total = total_distance(activities) # Write output into the
Element("output").write(f"Total Distance: {total} km")
