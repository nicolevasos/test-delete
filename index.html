My Simple Strava Dashboard

# Simulated Strava activities activities = [ {"name": "Morning Run", "distance": 5}, {"name": "Evening Ride", "distance": 20}, {"name": "Hike", "distance": 8} ] # Function to calculate total distance def total_distance(activities): return sum(a["distance"] for a in activities) # Display the result total = total_distance(activities) print(f"Total Distance: {total} km")
