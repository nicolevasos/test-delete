from pyscript import Element activities = [ {"name": "Morning Run", "distance": 5}, {"name": "Evening Ride", "distance": 20}, {"name": "Hike", "distance": 8} ] def total_distance(activities): return sum(a["distance"] for a in activities) total = total_distance(activities) # Inject into
Element("output").write(f"Total Distance: {total} km")
